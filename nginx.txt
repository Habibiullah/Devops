server {
    server_name uatapp.pharmalyfe.com;
    
    location / {
        proxy_pass http://localhost:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/uatapp.pharmalyfe.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/uatapp.pharmalyfe.com/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;
}

server {
    server_name uatapi.pharmalyfe.com;

    location /api/ {
        proxy_pass http://localhost:5010;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/uatapi.pharmalyfe.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/uatapi.pharmalyfe.com/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;
}

server {
    if ($host = uatapp.pharmalyfe.com) {
        return 301 https://$host$request_uri;
    }
    listen 80;
    server_name uatapp.pharmalyfe.com uatapi.pharmalyfe.com;
    return 404; # managed by Certbot
}
